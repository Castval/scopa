== DAL TUO PC ==

rsync -avz --exclude node_modules --exclude "scopa.db*" --exclude "logs" -e "ssh -i ~/Downloads/ssh-key-2026-04-08.key" ~/dev/scopa/ ubuntu@130.110.19.57:~/scopa/


== CONNESSIONE SSH ==

ssh -i ~/Downloads/ssh-key-2026-04-08.key ubuntu@130.110.19.57


== AGGIORNAMENTO (dopo rsync) ==

cd ~/scopa
npm install
# Riavvia con --update-env per riapplicare modifiche a ecosystem.config.js
# (max_memory_restart, node_args, env). Senza --update-env tiene la config vecchia.
pm2 restart ecosystem.config.js --update-env


== PRIMO AVVIO (solo la prima volta) ==

cd ~/scopa
mkdir -p logs
npm install
pm2 start ecosystem.config.js
pm2 save
pm2 startup   # segui le istruzioni stampate per abilitare avvio automatico


== MIGRAZIONE DA AVVIO VECCHIO ==
# Se in passato avevi avviato con 'pm2 start server.js --name scopa', il processo
# non aveva max_memory_restart ne' heap cap. Per applicare la config nuova:

cd ~/scopa
pm2 delete scopa
pm2 start ecosystem.config.js
pm2 save


== CONFIGURAZIONE NGINX (solo la prima volta) ==

sudo tee /etc/nginx/sites-available/default << 'EOF'
server {
    listen 80;
    server_name scopa.duckdns.org;

    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }
}
EOF
sudo nginx -t && sudo systemctl restart nginx


== VERIFICA ==

sudo systemctl status nginx
pm2 status
curl http://localhost:3000/health


== HTTPS CON CERTBOT (opzionale) ==

sudo apt install -y certbot python3-certbot-nginx
sudo certbot --nginx -d scopa.duckdns.org
